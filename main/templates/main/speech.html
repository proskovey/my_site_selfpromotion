Телеграмм бот (продажа авто)

        

            import telebot
            from telebot import types
            bot = telebot.TeleBot('TOKEN')

            @bot.message_handler(commands=['start'])
            def start(message):
                markup = types.ReplyKeyboardMarkup(resize_keyboard=True)
                btn1 = types.KeyboardButton('Начнём')
                markup.add(btn1)
                bot.send_message(message.from_user.id, 'AVTOUsed', reply_markup=markup)

            @bot.message_handler(content_types=['text'])
            def get_text_message(message):

                if message.text == 'Начнём':
                    markup = types.ReplyKeyboardMarkup(resize_keyboard=True)
                    btn1 = types.KeyboardButton('Mazda 6')
                    btn2 = types.KeyboardButton('Volkswagen Tiguan')
                    btn3 = types.KeyboardButton('Контакты')
                    markup.add(btn1, btn2, btn3)
                    bot.send_message(message.from_user.id, 'Выбери авто', reply_markup=markup)

                elif message.text == 'Mazda 6':
                    bot.send_photo(message.from_user.id, 'https://avatars.mds.yandex.net/get-autoru-vos
                    /2111725/d061b06cb252615db219d3edd1cd591c/1200x900', caption='Описание автомобиля. Цена. Телефон')

                elif message.text == 'Volkswagen Tiguan':
                    bot.send_photo(message.from_user.id, 'https://i.pinimg.com/originals/49/9f/a1/
                    499fa104a723756b1b18767671ce9ffe.jpg', caption='Описание автомобиля. Цена. Телефон')

                elif message.text == 'Контакты':
                    bot.send_message(message.from_user.id, text='Телефон: +79999999999, Email: 1234567890.@gmail.com')

            bot.polling(none_stop=True, interval=0)
